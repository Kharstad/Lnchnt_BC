instalação{
	npm i				instala as dependencias do angular (pacotes e modulos)
	npm i -g @pacote/..		Instala um pacote especifico
}

server JSON {
	npm i -g json			Instala o servidor
	npm s -o			abre o servidor no caminho do hd
	json-server --watch db.json	Inicia o servidor especificado no fim do cmdlet
}
server IONIC {
	ionic s 			Inicia a aplicação. 
}
